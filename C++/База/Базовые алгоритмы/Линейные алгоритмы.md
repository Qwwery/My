## Задача 1:
есть массив из 99 элементов (a1, a2, ..., a99)
1 <= ai <= 100
ai != aj
найти элемент, которого не хватает
### Наивное решение (O(n ^ 2))
проходим массив двумя циклами фор

```
int n;
cin >> n;
bool flag = false;
for (int i = 1; i <= 100; i++)
	for (int j; j <= 99; j++)
		if (a[j] == i)
			flag = true;
if (flag)
	cout << "YES";
else 
	cout << "NO";
```
### Решение через сумму
```
int sum = 0;
sum = 100 (100 + 1) / 2;
for (int i = 0; i < 99; i++) 
	sum -= a[i];
cout << sum;
```
## Задача 2:
Есть число n
массив длинны n: a1, a2, ..., an
число k
Найти отрезок, на котором будет максимальная сумма 
Решение наивное:
```
int maxs = 0, sum;
...
for (int i = 0; i < n; i++) {
	sum = 0;
	for (int j = 0; j < k; j++) {
		if (i + k < n) {
			sum += A[i + k];
		}
	}
	if (maxs < sum)
		sum = maxs;
}
cout << maxs;
```
Нормальное решение:
```
int n, k;
cin >> n >> k;
vector<int> A(n);
for (int i = 0; i < n; i++)
	cin >> A[i];
int sumn = 0;
for (int i = 0; i < k; i++) 
	sumn += A[i];
int sum_max = sumn;
for (int i = k; i < n; i++) {
	sumn += A[i] - A[i - k];
	if (sumn > sum_max)
		sum_max = sumn;
}
	cout << sum_max;
```

## Задача 3:
Найти максимальное и минимальное значение в массиве за O(3 * n)
```
int n;
cin >> n;
vector<int> A(n);
for (int i = 0; i < n; i++)
	cin >> A[i];
vector<int> MinArr;
vector<int> MaxArr;
for (int i = 0; i < n; i += 2) {
	if (A[i] > A[i + 1]) {
		MaxArr.push_back(A[i]);
		MinArr.push_back(A[i + 1]);
	}
	else {
		MinArr.push_back(A[i]);
		MaxArr.push_back(A[i + 1]);
	}
}
int min_el = MinArr[0], max_el = MaxArr[0];
for (int i = 0; i < MinArr.size(); i++) {
	min_el = min(min_el, MinArr[i]);
	max_el = max(max_el, MaxArr[i]);
}
cout << min_el << ' ' << max_el;
```
